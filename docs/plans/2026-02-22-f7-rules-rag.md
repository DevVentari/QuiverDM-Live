# Feature 7: Rules/Source Companion (Local RAG)

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Inline rules lookup in session view — DM types a question, gets an answer with source citations from SRD and uploaded rule PDFs.

**Architecture:** Reuses pgvector `Embedding` model from Feature 2 (`entityType='rules'`, `campaignId=null`). Rules PDFs are marked `isRulesSource=true` in `HomebrewPDF` and re-chunked/embedded. SRD loaded from a bundled JSON seed. New `rules` tRPC router. Floating rules panel in session view. Redis caches answers (10-min TTL).

**Dependencies:** Feature 2 must be merged first (provides `Embedding` model + `embedding.repository.ts` + `embeddings.ts`). Ensure those files exist before starting.

**Tech Stack:** Prisma + pgvector, Redis, Ollama (LLM + embeddings), tRPC, Next.js App Router, shadcn/ui

## Task 2: Rules Indexing Service

**Files:**
- Create: `src/server/services/rules-indexing.service.ts`

**Step 1: Create service**

```typescript
import { prisma } from '../db';
import { generateEmbedding, chunkText } from '@/lib/ai/embeddings';
import { upsertEmbeddings } from '@/server/repositories/embedding.repository';

/**
 * Index a PDF as a rules source.
 * Expects the PDF to already have extracted text (via homebrewPDF.extractedText or rawText).
 */
export async function indexRulesSource(pdfId: string): Promise<void> {
  const pdf = await prisma.homebrewPDF.findUnique({
    where: { id: pdfId },
    select: { id: true, filename: true, extractedText: true },
  });
  if (!pdf) throw new Error(`PDF not found: ${pdfId}`);

  const text = (pdf as any).extractedText;
  if (!text) throw new Error(`PDF ${pdfId} has no extracted text — process it first`);

  const chunks = chunkText(text, 1500, 150);
  const embeddedChunks = await Promise.all(
    chunks.map(async (chunk, index) => ({
      text: chunk,
      index,
      vector: await generateEmbedding(chunk),
    }))
  );

  await upsertEmbeddings(
    pdfId,
    'rules',
    embeddedChunks,
    { source: pdf.filename, pdfId },
    undefined // no campaignId — global
  );

  await prisma.homebrewPDF.update({
    where: { id: pdfId },
    data: { isRulesSource: true, indexedAt: new Date() },
  });

  console.log(`[rules] Indexed ${embeddedChunks.length} chunks from ${pdf.filename}`);
}

/**
 * Remove rules index for a PDF.
 */
export async function removeRulesIndex(pdfId: string): Promise<void> {
  await prisma.embedding.deleteMany({ where: { entityId: pdfId, entityType: 'rules' } });
  await prisma.homebrewPDF.update({
    where: { id: pdfId },
    data: { isRulesSource: false, indexedAt: null },
  });
}
```

```typescript
// src/server/services/rules.service.ts
import { generateEmbedding } from '@/lib/ai/embeddings';
import { prisma } from '../db';
import { chatWithOllama } from '@/lib/ai/ollama';
import { redis } from '@/lib/queue/queue';
import { indexRulesSource, removeRulesIndex } from './rules-indexing.service';

export class RulesService {
  async lookup(question: string, limit = 5): Promise<{
    answer: string;
    sources: Array<{ chunkText: string; source: string }>;
  }> {
    // Check Redis cache
    const cacheKey = `rules:${Buffer.from(question).toString('base64').slice(0, 40)}`;
    if (redis) {
      const cached = await redis.get(cacheKey);
      if (cached) return JSON.parse(cached);
    }

    // Embed the question
    const queryVector = await generateEmbedding(question);

    // Search rules embeddings (no campaignId filter)
    const results = await prisma.$queryRaw<any[]>`
      SELECT
        "entityId",
        "chunkText",
        metadata,
        1 - (vector <=> ${`[${queryVector.join(',')}]`}::vector) AS score
      FROM "Embedding"
      WHERE "entityType" = 'rules'
      ORDER BY vector <=> ${`[${queryVector.join(',')}]`}::vector
      LIMIT ${limit}
    `;

    if (results.length === 0) {
      return { answer: 'No relevant rules found for that question.', sources: [] };
    }

    const context = results.map((r: any) => r.chunkText).join('\n\n---\n\n');

    const response = await chatWithOllama([
      {
        role: 'system',
        content: 'You are a D&D 5e rules expert. Answer the question using ONLY the provided rules text. Be concise (2-4 sentences). If the answer is not in the provided text, say so.',
      },
      {
        role: 'user',
        content: `Rules text:\n${context}\n\nQuestion: ${question}`,
      },
    ], { temperature: 0.1 });

    const result = {
      answer: response.message.content,
      sources: results.map((r: any) => ({
        chunkText: r.chunkText.slice(0, 200),
        source: r.metadata?.source ?? 'Unknown source',
      })),
    };

    // Cache for 10 minutes
    if (redis) {
      await redis.setex(cacheKey, 600, JSON.stringify(result));
    }

    return result;
  }

  async listSources() {
    return prisma.homebrewPDF.findMany({
      where: { isRulesSource: true },
      select: { id: true, filename: true, indexedAt: true },
      orderBy: { indexedAt: 'desc' },
    });
  }

  async listAllPdfs() {
    return prisma.homebrewPDF.findMany({
      select: { id: true, filename: true, isRulesSource: true, indexedAt: true },
      orderBy: { createdAt: 'desc' },
    });
  }

  async indexSource(pdfId: string) {
    await indexRulesSource(pdfId);
  }

  async removeSource(pdfId: string) {
    await removeRulesIndex(pdfId);
  }
}

export const rulesService = new RulesService();
```

**Step 2: Create rules router**

```typescript
// src/server/routers/rules.ts
import { router, protectedProcedure } from '../trpc';
import { z } from 'zod';
import { rulesService } from '../services/rules.service';
import { requireAdmin } from './feedback'; // reuse admin check, or inline it

export const rulesRouter = router({
  lookup: protectedProcedure
    .input(z.object({
      question: z.string().min(3).max(500),
      limit: z.number().int().min(1).max(10).optional().default(5),
    }))
    .query(({ input }) =>
      rulesService.lookup(input.question, input.limit)
    ),

  listSources: protectedProcedure
    .query(() => rulesService.listSources()),

  listAllPdfs: protectedProcedure
    .query(() => rulesService.listAllPdfs()),

  indexSource: protectedProcedure
    .input(z.object({ pdfId: z.string() }))
    .mutation(async ({ input, ctx }) => {
      // Admin only — check ADMIN_EMAILS
      const adminEmails = (process.env.ADMIN_EMAILS ?? '').split(',').map((e) => e.trim());
      if (!adminEmails.includes(ctx.session.user.email ?? '')) {
        throw new TRPCError({ code: 'FORBIDDEN', message: 'Admin only' });
      }
      await rulesService.indexSource(input.pdfId);
      return { success: true };
    }),

  removeSource: protectedProcedure
    .input(z.object({ pdfId: z.string() }))
    .mutation(async ({ input, ctx }) => {
      const adminEmails = (process.env.ADMIN_EMAILS ?? '').split(',').map((e) => e.trim());
      if (!adminEmails.includes(ctx.session.user.email ?? '')) {
        throw new TRPCError({ code: 'FORBIDDEN', message: 'Admin only' });
      }
      await rulesService.removeSource(input.pdfId);
      return { success: true };
    }),
});
```

```tsx
'use client';

import { useState } from 'react';
import { trpc } from '@/lib/trpc';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Sheet, SheetContent, SheetHeader, SheetTitle, SheetTrigger } from '@/components/ui/sheet';
import { Card, CardContent } from '@/components/ui/card';
import { BookOpen, Search, Loader2, ChevronDown, ChevronUp } from 'lucide-react';

export function RulesPanel() {
  const [question, setQuestion] = useState('');
  const [submittedQuestion, setSubmittedQuestion] = useState('');
  const [expandedSources, setExpandedSources] = useState(false);

  const { data: result, isFetching } = trpc.rules.lookup.useQuery(
    { question: submittedQuestion },
    { enabled: submittedQuestion.length >= 3 }
  );

  return (
    <Sheet>
      <SheetTrigger asChild>
        <Button variant="outline" size="sm" className="fixed bottom-4 right-4 shadow-lg z-10">
          <BookOpen className="h-4 w-4 mr-2" /> Rules
        </Button>
      </SheetTrigger>
      <SheetContent side="right" className="w-96 overflow-y-auto">
        <SheetHeader>
          <SheetTitle className="flex items-center gap-2">
            <BookOpen className="h-4 w-4" /> Rules Companion
          </SheetTitle>
        </SheetHeader>

        <div className="space-y-4 mt-4">
          <div className="flex gap-2">
            <Input
              placeholder="Ask a rules question…"
              value={question}
              onChange={(e) => setQuestion(e.target.value)}
              onKeyDown={(e) => e.key === 'Enter' && setSubmittedQuestion(question)}
            />
            <Button
              size="sm"
              onClick={() => setSubmittedQuestion(question)}
              disabled={question.length < 3 || isFetching}
            >
              {isFetching ? <Loader2 className="h-4 w-4 animate-spin" /> : <Search className="h-4 w-4" />}
            </Button>
          </div>

          {isFetching && (
            <div className="flex items-center gap-2 text-sm text-muted-foreground">
              <Loader2 className="h-4 w-4 animate-spin" />
              Looking up rules…
            </div>
          )}

          {result && !isFetching && (
            <div className="space-y-3">
              <Card>
                <CardContent className="pt-4">
                  <p className="text-sm">{result.answer}</p>
                </CardContent>
              </Card>

              {result.sources.length > 0 && (
                <div>
                  <Button
                    variant="ghost"
                    size="sm"
                    className="text-xs text-muted-foreground p-0 h-auto"
                    onClick={() => setExpandedSources((v) => !v)}
                  >
                    {expandedSources ? <ChevronUp className="h-3 w-3 mr-1" /> : <ChevronDown className="h-3 w-3 mr-1" />}
                    {result.sources.length} source{result.sources.length !== 1 ? 's' : ''}
                  </Button>

                  {expandedSources && (
                    <div className="space-y-2 mt-2">
                      {result.sources.map((s: any, i: number) => (
                        <div key={i} className="text-xs border rounded p-2 bg-muted/30">
                          <p className="font-medium text-muted-foreground mb-1">from: {s.source}</p>
                          <p className="line-clamp-3">{s.chunkText}…</p>
                        </div>
                      ))}
                    </div>
                  )}
                </div>
              )}
            </div>
          )}

          <div className="space-y-1">
            <p className="text-xs text-muted-foreground font-medium">Try asking:</p>
            {['How does grappling work?', 'What conditions cause disadvantage?', 'How is spell save DC calculated?'].map((q) => (
              <button
                key={q}
                className="block text-xs text-primary hover:underline text-left"
                onClick={() => { setQuestion(q); setSubmittedQuestion(q); }}
              >
                {q}
              </button>
            ))}
          </div>
        </div>
      </SheetContent>
    </Sheet>
  );
}
```

## Task 6: Admin Rules Sources Page

**Files:**
- Create: `src/app/(app)/admin/rules-sources/page.tsx`

**Step 1: Create admin page**

```tsx
'use client';

import { trpc } from '@/lib/trpc';
import { Button } from '@/components/ui/button';
import { Badge } from '@/components/ui/badge';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { BookOpen, Plus, Minus, Loader2 } from 'lucide-react';
import { toast } from 'sonner';
import { format } from 'date-fns';

export default function RulesSourcesPage() {
  const utils = trpc.useUtils();
  const { data: pdfs, isLoading } = trpc.rules.listAllPdfs.useQuery();

  const indexMutation = trpc.rules.indexSource.useMutation({
    onSuccess: () => { utils.rules.listAllPdfs.invalidate(); toast.success('PDF indexed as rules source'); },
    onError: (e) => toast.error(e.message),
  });
  const removeMutation = trpc.rules.removeSource.useMutation({
    onSuccess: () => { utils.rules.listAllPdfs.invalidate(); toast.success('Removed from rules sources'); },
    onError: (e) => toast.error(e.message),
  });

  if (isLoading) return <div className="p-6 animate-pulse">Loading…</div>;

  return (
    <div className="p-6 max-w-2xl mx-auto space-y-6">
      <div className="flex items-center gap-2">
        <BookOpen className="h-5 w-5" />
        <h1 className="text-2xl font-bold">Rules Sources</h1>
      </div>
      <p className="text-sm text-muted-foreground">
        Mark PDFs as rules sources to index them for the in-session Rules Companion lookup.
        The PDF must have been processed (text extracted) before indexing.
      </p>

      <div className="space-y-3">
        {pdfs?.map((pdf: any) => (
          <Card key={pdf.id}>
            <CardContent className="pt-4 flex items-center justify-between">
              <div>
                <p className="font-medium text-sm">{pdf.filename}</p>
                {pdf.indexedAt && (
                  <p className="text-xs text-muted-foreground">Indexed {format(new Date(pdf.indexedAt), 'MMM d, yyyy')}</p>
                )}
              </div>
              <div className="flex items-center gap-2">
                {pdf.isRulesSource && <Badge variant="secondary">Rules Source</Badge>}
                {pdf.isRulesSource ? (
                  <Button size="sm" variant="outline" onClick={() => removeMutation.mutate({ pdfId: pdf.id })} disabled={removeMutation.isPending}>
                    <Minus className="h-3 w-3 mr-1" /> Remove
                  </Button>
                ) : (
                  <Button size="sm" onClick={() => indexMutation.mutate({ pdfId: pdf.id })} disabled={indexMutation.isPending}>
                    {indexMutation.isPending ? <Loader2 className="h-3 w-3 animate-spin mr-1" /> : <Plus className="h-3 w-3 mr-1" />}
                    Index
                  </Button>
                )}
              </div>
            </CardContent>
          </Card>
        ))}
        {pdfs?.length === 0 && <p className="text-sm text-muted-foreground">No PDFs uploaded yet.</p>}
      </div>
    </div>
  );
}
```
